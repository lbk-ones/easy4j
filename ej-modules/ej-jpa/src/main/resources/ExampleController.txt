package easy4j.module.jpa.base;

import org.springframework.web.bind.annotation.PostMapping;
import org.springframework.web.bind.annotation.RequestBody;
import org.springframework.web.bind.annotation.RequestMapping;

import javax.annotation.Resource;
import java.util.Map;
/**
 * 这个是拿来复制的
 * @author bokun.li
 * @date 2023/6/2
 */
//@RestController
@RequestMapping("example")
public class ExampleController extends BaseController {

    @Resource
    IXxxService iXxxService;

    // 保存
    @PostMapping("save")
    public EasyResult<Object> save(@RequestBody XXXDto object) throws AppException {
        iXxxService.save(object);
        return R.ok();
    }

    // 删除
    @PostMapping("deleteByIds")
    public EasyResult<Object> deleteByIds(@RequestBody XXXDto object) throws AppException{
        iXxxService.deleteByIds(object);
        return R.ok();
    }

    // 更改
    @PostMapping("update")
    public EasyResult<Object> update(@RequestBody XXXDto object) throws AppException{
        XXXDto XXXDto = iXxxService.update(object);
        return R.ok(sysAppsDto);
    }
    // 分页查询
    @PostMapping("refreshPage")
    public EasyResult<Object> refreshPage(@RequestBody PageParamsDto pageParamsDto) throws AppException{
        Map<String,Object> objectMap = iXxxService.refreshPage(pageParamsDto);
        return R.ok(objectMap);
    }

    // 启用 禁用
    @PostMapping("enable")
    public EasyResult<Object> enable(@RequestBody XXXDto object) throws AppException{
        String msg = iXxxService.enable(object);
        return R.ok(msg);
    }

    // 根据id查询
    @PostMapping("getById")
    public EasyResult<Object> getById(@RequestBody XXXDto object) throws AppException{
        XXXDto byId = iXxxService.getById(object);
        return R.ok(byId);
    }
}
